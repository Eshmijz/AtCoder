ここの遷移サンプル

   rep(i, n) rep(j, K + 1) rep(k, 2) rep(x, 10) {
       int num = N[i] - '0';
       int ni = i + 1, nj = j + (x != 0), nk = k | (x < num);
       if(x > num && k == 0) continue;
       if(nj > K) continue;
       dp[ni][nj][nk] += dp[i][j][k];
   }
